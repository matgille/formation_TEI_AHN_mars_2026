\section{Organisation de la TEI}

\subsection{Éléments et attributs}

\begin{frame}{Éléments proposés par la TEI}
\begin{itemize}
\item La TEI propose d'utiliser un certain nombre d'éléments définis en amont. 
\item Faire de la TEI, c'est utiliser ce jeu d'éléments pour décrire un objet textuel.
\item Exemples: l'élément \texttt{div} est utilisé pour indiquer une section, soit une division structurelle du texte. L'élément \texttt{p} est utilisé pour identifier des paragraphes dans des textes en prose. 
\end{itemize}
\end{frame}

\begin{frame}[containsverbatim]{Représenter l'incipit de la \textit{Recherche}}
La \textit{Recherche} est structurée en tomes, eux-même divisés en parties et en chapitres.
\begin{minted}[breaklines,numbersep=10mm]{xml}
<div>
    <div>
        <p>Longtemps, je me suis couché de bonne heure 
        [...]
        </p>
    </div>
</div>
\end{minted}
\end{frame}

\begin{frame}{Attributs proposés par la TEI}
\begin{itemize}
\item Chaque élément peut être précisé par un attribut.
\item Ces attributs (sauf ceux qui commencent par \texttt{`xml:`}) sont également spécifiques à la TEI
\item Exemple: \texttt{n} est un attribut permettant de compter un élément (un vers, par exemple, ou un chapitre)

\end{itemize}
\end{frame}

\begin{frame}[containsverbatim]{L'utilisation d'attributs propres à la TEI}
\begin{minted}[breaklines,numbersep=10mm]{xml}
<div n="1">
    <div n="1">
        <p>Longtemps, je me suis couché de bonne heure 
        [...]
        </p>
    </div>
</div>
\end{minted}
\end{frame}

\begin{frame}{Attributs proposés par la TEI}
\begin{itemize}
\item Nous avons souvent besoin de préciser et de classer les différents éléments
\item La TEI propose un attribut \texttt{type} pour ce faire
\end{itemize}
\end{frame}

\begin{frame}[containsverbatim]{L'utilisation d'attributs propres à la TEI}
\begin{minted}[breaklines,numbersep=10mm]{xml}
<div n="1" type="partie">
    <div n="1" type="chapitre">
        <p>Longtemps, je me suis couché de bonne heure 
        [...]
        </p>
    </div>
</div>
\end{minted}
\end{frame}

\subsection{Des objets définis et illustrés}
\begin{frame}
\begin{itemize}
\item La TEI ne se limite pas à proposer une liste d'éléments et d'attributs
\item Il est nécessaire de les définir afin de favoriser leur utilisation
\item Chaque élément et chaque attribut est ainsi défini dans les Guidelines
\item Taper \enquote{\texttt{TEI element}} sur votre moteur de recherche favori vous mène directement à la définition de l'élément \texttt{element} cherché.
\end{itemize}
\end{frame}


\begin{frame}{Exemple: \texttt{p}}
\begin{itemize}
\item L'élément p est défini ici: \href{https://www.tei-c.org/release/doc/tei-p5-doc/fr/html/ref-p.html}{https://www.tei-c.org/release/doc/tei-p5-doc/fr/html/ref-p.html}
\end{itemize}
\end{frame}

\begin{frame}{Ce que contient une page décrivant un élément}
La page contient: 
\begin{enumerate}
\item Une définition avec un renvoi vers le \textbf{module} correspondant (voir plus bas)
\item La liste des attributs autorisés
\item La définition selon le modèle de la TEI 
\item Les éléments qui peuvent contenir l'élément courant
\item Les éléments contenus dans l'élément courant
\item Un exemple et un lien vers une liste d'exemples d'utilisation de l'élément
\end{enumerate}
\end{frame}


\begin{frame}{Exemple: \texttt{p}}
\begin{enumerate}
\item Il est nécessaire d'aller lire la définition d'un élément avant de l'utiliser
\item Il est quasi-certain qu'il existe déjà un élément pour votre besoin
\item Ne réinventez pas la roue !
\end{enumerate}
\end{frame}

\subsection{Les chapitres thématiques de la TEI}

\begin{frame}
\begin{itemize}
\item Les \textit{Guidelines} sont organisés en chapitres qui viennent décrire un thème spécifique
\item Chaque chapitre est décrit en langage naturel et vient présenter des cas typiques d'encodage.
\item Avant de commencer un projet, il faut avoir lu les chapitres correspondants afin de faire les meilleurs choix possibles.
\item Chaque chapitre est lié à un \textbf{module} spécifique (je reviendrai dessus plus loin)
\end{itemize}
\end{frame}

\subsection{Chapitres fondamentaux}

\subsubsection{Métadonnées}

\begin{frame}{Les métadonnées et le \texttt{teiHeader}}
\begin{itemize}
\item Un document XML-TEI est composé au minimum de deux parties: les métadonnées et l'objet décrit.
\item Les métadonnées, qui décrivent le document en lui-même (auteur.e, source, régime de publication, année, etc), sont renseignées dans la première partie, le \texttt{teiHeader}
\item Nous allons voir les éléments \texttt{fileDesc} et \texttt{encodingDesc}, mais il en existe d'autres
\end{itemize}
\end{frame}


\begin{frame}{\texttt{fileDesc}}
\begin{itemize}
\item Le \texttt{fileDesc} contient toutes les informations sur le fichier encodé: son titre, 
son régime de distribution (licence), la source, etc. 
\end{itemize}
\end{frame}


\begin{frame}[containsverbatim]{\texttt{fileDesc} > \texttt{titleStmt}}
\begin{itemize}
\item Le \texttt{titleStmt} contient l'information sur le titre \textbf{du document XML}, qui peut différer du titre de la ou des sources utilisées.
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<fileDesc>
    <titleStmt>
      <title>The shortest possible TEI document</title>
    </titleStmt>
    ...
</fileDesc>
\end{minted}
\end{frame}

\begin{frame}[containsverbatim]{\texttt{fileDesc} > \texttt{titleStmt}}
\begin{itemize}
\item Le \texttt{titleStmt} contient également toute information sur l'origine et la création du document TEI décrit: autorité, traduction, édition, financements, responsabilité intellectuelle (en cas de multiples auteurs).
\item Éléments importants: \texttt{respStmt} \texttt{resp}, \texttt{name}, \texttt{author}, \texttt{editor}, etc 
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<fileDesc>
    <titleStmt>
      <title>The shortest possible TEI document</title>
    </titleStmt>
    <respStmt>
      <resp>created by</resp>
      <name>Ann O'Nyme</name>
    </respStmt>
</fileDesc>
\end{minted}
\end{frame}

\begin{frame}[containsverbatim]{\texttt{fileDesc} > \texttt{publicationStmt}}
\begin{itemize}
\item Le \texttt{publicationStmt} donne des informations sur le contexte de publication et de diffusion du document (propriété intellectuelle essentiellement)
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<publicationStmt>
    <distributor>ATILF (Analyse et Traitement Informatique de la Langue Française)</distributor>
    <idno type="FRANTEXT">L434</idno>
    <address>
        <addrLine>44, avenue de la Libération</addrLine>
        ...
        <addrLine>FRANCE</addrLine>
    </address>
    <availability status="free">
        <p>Dans un cadre de recherche ou d'enseignement</p>
    </availability>
</publicationStmt>
\end{minted}
\end{frame}


\begin{frame}[containsverbatim]{\texttt{fileDesc} > \texttt{sourceDesc}}
\begin{itemize}
\item L'élément \texttt{sourceDesc} permet de décrire la source utilisée pour l'encodage (manuscrit, audio, livre imprimé). 
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<sourceDesc>
    <bibl>
        The first folio of Shakespeare, prepared by Charlton Hinman (The Norton Facsimile, 1968)
    </bibl>
</sourceDesc>
\end{minted}
\end{frame}



\begin{frame}[containsverbatim]{\texttt{encodingDesc}}
\begin{itemize}
\item L'\texttt{encodingDesc} contient des informations sur la façon d'encoder le document en cours. Cette description sera réalisée en language naturel, au sein d'un \texttt{editorialDecl} puis d'un paragraphe.
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<encodingDesc>
    <editorialDecl>
        <p>
            Basic encoding, capturing lexical information only. All hyphenation, punctuation, and variant spellings normalized. No formatting or layout information preserved.
        </p>
    </editorialDecl>
</encodingDesc>
\end{minted}
\end{frame}


\begin{frame}[containsverbatim]{\texttt{encodingDesc} > \texttt{projectDesc}}
\begin{itemize}
\item L'\texttt{encodingDesc} peut contenir également un élément \texttt{projectDesc}. Cet élément vient décrire le cadre de l'encodage du fichier.
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<encodingDesc>
    <projectDesc>
        <p>
            Ce document contient l'ensemble des fichiers XML de ma thèse de doctorat, réalisé sous la direction de Carlos Heusch (Ens de Lyon) et de Jesús R. Velasco (Yale University), soutenue le 8 septembre 2023.
        </p>
    </projectDesc>
<encodingDesc>
\end{minted}
\end{frame}


\begin{frame}[containsverbatim]{\texttt{encodingDesc} > \texttt{projectDesc}}
\begin{itemize}
\item On peut combiner les deux:
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<encodingDesc>
    <projectDesc>
        <p>Description du projet</p>
    </projectDesc>
    <editorialDecl>
        <p>Description des normes d'encodage et de transcription</p>
    </editorialDecl>
<encodingDesc>
\end{minted}
\end{frame}


\begin{frame}{Exercice: encoder les metadonnées du \textit{Dormeur du val}}
\begin{itemize}
\item Proposez un encodage minimal des metadonnées du poème de Rimbaud. Vous êtes responsable
de l'édition du poème et de son encodage. Proposez une licence de diffusion libre (Creative Commons par exemple) de votre édition.
\end{itemize}
\end{frame}



\subsubsection{Structures textuelles}

\begin{frame}{Structure textuelle}
\begin{itemize}
\item Les divisions sont marquées par \texttt{div} ou \texttt{div1, div2, div3}. Le choix entre les deux est obligatoire !
\item Les paragraphes de la prose, par \texttt{p}
\item Les sections moins marquées sémantiquement peuvent être indiquées par \texttt{ab} (anonymous block)
\end{itemize}
\end{frame}


\begin{frame}{Structure textuelle: attributs}
\begin{itemize}
\item Dans le cas du choix des éléments \texttt{div} no numérotés, il est important de préciser le niveau de structure à l'aide d'attributs
\item On utilisera d'une part l'attribut \texttt{@type} pour la classification du niveau structurel
\item On utilisera de l'autre l'attribut \texttt{@n} pour la numéro de la structure courante
\item Il est parfois utile de créer des identifiants uniques pour chaque unité structurelle, afin de pouvoir la retrouver facilement ensuite. On utilise alors l'attribut xml \texttt{@xml:id}
\item Attention, \textbf{@xml:id n'accepte que des valeurs uniques dans un même document}
\end{itemize}
\end{frame}

\begin{frame}{Exercice}
\begin{itemize}
\item Dans un nouveau document XML, encodez la structure minimale de l'incipit de la \textit{Recherche}, comme si vous deviez éditer le texte intégral.
\end{itemize}
\end{frame}

\subsubsection{Description de manuscrits}
\begin{frame}{Description de manuscrits}
\begin{itemize}
\item Ce module peut ête confondu avec le module précédent, mais il concerne la description physique du manuscrit et non pas la description de la page.

\subsubsection{Entités nommées}
\begin{frame}{Entités nommées}
\begin{itemize}
\item Une entité nommée est une chaîne de caractères qui décrit un référent humain, géographique ou institutionnel (personne, lieu, entreprise, organisation, etc). 
\item La TEI propose un ensemble d'éléments permettant d'identifier les entités nommées dans le texte, et de les désambiguiser (dans le teiHeader).
\end{itemize}
\end{frame}


\begin{frame}{Identifier les entités nommées}
\begin{itemize}
\item 
\end{itemize}
\end{frame}

\begin{frame}{Désambiguiser les entités nommées}
\begin{itemize}
\item On a besoin d'indiquer le référent de chaque entité, ce que l'on peut faire en plusieurs points de l'édition
\item La plus commune est dans le \texttt{teiHeader}
\item Il s'agit de proposer une liste de personnes (\texttt{listPerson}), ou de lieux (\texttt{listPlace})
\item Ces deux éléments seront présents dans l'élément \texttt{sourceDesc}
\end{itemize}
\end{frame}

\section{Application: encoder des vers et du théâtre}

\subsection{Le vers}

\begin{frame}[containsverbatim]{Les formes versifiées}
\begin{itemize}
\item Le module de la TEI est: \href{Verse}{https://tei-c.org/release/doc/tei-p5-doc/en/html/VE.html}
\item les éléments fondamentaux sont le vers (\texttt{l} pour \textit{line}), 
\item le groupe de vers (\texttt{lg} pour \textit{LineGroup})
\end{itemize}
\end{frame}


\begin{frame}[containsverbatim]{Les formes versifiées}
\begin{itemize}
\item Vous pouvez représenter une unité textuelle (poème) par comme un \texttt{lg} ou plus classiquement une division. 
\item L'hémistiche ou le pied peuvent être indiqués à l'aide de l'élément \texttt{seg} qui devra être précisé par un attribut \texttt{type}
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<lg type="chant" n="1">
    <lg>
        <l>Arma virumque cano, Troiae qui primus ab oris</l>
        <l>italiam, fato profugus, Laviniaque vēnit</l>
        <l>litora, multum ille et terris iactatus et alto</l>
        <l>vi superum saevae memorem Iunonis ob iram;</l>
        <l>multa quoque et bello passus, dum conderet urbem,</l>
        <l>inferretque deos Latio, genus unde Latinum,</l>
        <l>Albanique patrēs, atque altae moenia Romae.</l>
    </lg>
</lg>
\end{minted}
\end{frame}

\begin{frame}[containsverbatim]{Les formes versifiées}
\begin{itemize}
\item Vous pouvez décrire la rime avec l'attribut \texttt{rhyme}:


\begin{minted}[breaklines,numbersep=2mm]{xml}
<lg type="soneto">
    <lg rhyme="ABAB" type="cuarteto" n="1">
        <l>Tengo miedo a perder la maravilla</l>
        <l>de tus ojos de estatua y el acento</l>
        <l>que me pone de noche en la mejilla</l>
        <l>la solitaria rosa de tu aliento.</l>
    </lg>
    ...
</lg>
\end{minted}
\end{itemize}
\end{frame}

\begin{frame}[containsverbatim]{Les formes versifiées}
\begin{itemize}
\item Vous pouvez décrire la structure métrique avec \texttt{met}:


\begin{minted}[breaklines,numbersep=2mm]{xml}
<l met="+--|+--|++|++|+--|++">
Arma virumque cano Troiae qui primus ab oris
</l>
\end{minted}
ou encore 
\begin{minted}[breaklines,numbersep=2mm]{xml}
<l>
 <seg type="foot" met="+--">Arma vi</seg>
 <seg met="+--">rumque ca</seg>
 <seg met="++">no Tro</seg>
 <seg met="++">iae qui </seg>
 <seg met="+--">primus ab</seg>
 <seg met="++"> oris</seg>
</l>
\end{minted}
\end{itemize}
\end{frame}

\subsection{Le théâtre}

\begin{frame}[containsverbatim]{Le théâtre}
\begin{itemize}
\item Le module de la TEI est: \href{Performance texts}{https://tei-c.org/release/doc/tei-p5-doc/en/html/DR.html}
\item les éléments fondamentaux sont la division structurelle (similaire à \textit{supra}), les interventions des personnages (\texttt{sp})
\item L'indication de parole est réalisée à l'aide de \texttt{speaker}.
\end{itemize}
\begin{minted}[breaklines,numbersep=2mm]{xml}
<div type="scene" n="1">
    <head>Scène Première</head>
    <stage>Philinte, Alceste</stage>
    <sp>
        <speaker who="#philinte">Philinte</speaker>
        <l n="1">Qu'est-ce donc ? qu'avez-vous ?</l>
    </sp>
    <sp who="#alceste">
